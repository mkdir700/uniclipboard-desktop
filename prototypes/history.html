const syncClipboard = async (text) => {
  try {
    const response = await fetch('/api/sync', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ content: text, type: 'text' })
    });
    return await response.json();
  } catch (error) {
    console.error('同步失败:', error);
    return null;
  }
}
